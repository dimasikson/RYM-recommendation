      function cycleSearchType(event) {

         var searchTypes = new Array('a', 'l', 'y', 'b', 'j', 'r', 'v', 'g', 'F', 'A', 'u', 'p', 's');
         var len = searchTypes.length;
         var currentVal = $('#searchtype').val();
         var currentIndex = searchTypes.indexOf(currentVal);
         var nextIndex = currentIndex+1;
         var prevIndex = currentIndex-1;
         if ( nextIndex == len ) {
            nextIndex = 0;
         }
         if ( prevIndex < 0 ) {
            prevIndex = len-1;
         }

         var nextType = searchTypes[nextIndex];
         var prevType = searchTypes[prevIndex];


         switch (event.keyCode) {
            case 40: 
               selectSearchType(nextType);
               event.preventDefault();
            break;

            case 38:
               selectSearchType(prevType);
               event.preventDefault();
            break;

            case 13:
               $('#mainsearch')[0].submit();
               event.preventDefault();
            break;
            default: 
               
            break;
         }

        
         return false;

      }
